

// The ssymv function performs a symmetric matrix-vector multiplication.
// It multiplies a symmetric matrix A with a vector x and stores the result in vector y.

// Inputs:
int n = 3; // size of matrix A
float A[9] = {1, 2, 3, 2, 4, 5, 3, 5, 6}; // symmetric matrix A
float x[3] = {1, 2, 3}; // vector x
float y[3] = {0, 0, 0}; // vector y
// Function call:
cblas_ssymv(CblasRowMajor, CblasUpper, n, 1.0, A, n, x, 1, 0.0, y, 1);

// Expected output:
float y[3] = {14, 32, 50}; // vector y
